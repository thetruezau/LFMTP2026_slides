\begin{frame}
  \frametitle{System $\olb$}
  
  \begin{definition}[Syntax of $\olb$]
    \vspace{-1.5em} 
    \begin{align*}
      \text{(terms)} \hspace{-5em} &&\quad t,u &::= \lb x.t \mid
                                                 \only<1>{ \der{x}{l} }
                                                 \only<2->{ {\color{red} \der{x}{l}} }  \mid tl \\
      \text{(lists)} \hspace{-5em} &&\quad l   &::= \nil \mid \cons{u}{l}    
    \end{align*}
  \end{definition}
  
  \onslide<2->{ An unusual aspect of the syntax above is that the variable does not exist \emph{per se}. }

  \onslide<3->{ This forces the following substitution equation to create a cut ($tl$): }

  \only<3>{ \[ [u/x]({ \color{red} \der{x}{l} }) = u([u/x]l). \] }
  \onslide<4->{
    \[ \xymatrix@C=0pt{
        [& u & / & x &]({ \color{red} \der{x}{l} }) = u([u/x]l). \\
        & \text{ term } \ar@{.}@/^1pc/[u] \ar@{.}@/_1pc/[u] &
        & \text{ not term } \ar@{.}@/^1pc/[u] \ar@{.}@/_1pc/[u] &        
      }
    \]
  }
  
  % \vspace{-1em}
  % \onslide<4->{ Also notice that a substitution $\sub ux$ is replacing a variable $x$ which does not belong to the class of terms. }

\end{frame}
