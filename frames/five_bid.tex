\begin{frame}
  \frametitle{System $\ulb$}
  
  \begin{definition}[Syntax of $\ulb$]
    \vspace{-1em} 
    \begin{align*}
      \text{(terms)} \hspace{-5em} &&\quad P,Q      &::= \lb x.t \mid \app{E} \\
      \text{(eliminations)} \hspace{-5em} &&\quad E &::= \only<1>{ x } \only<2->{ {\color{red} x} } \mid \hd{P} \mid EQ
    \end{align*}
    \vspace{-1em}
  \end{definition}
  
  \onslide<2->{ Here, the variable lives in the elimination class and is not a term. }

  \onslide<3->{ Therefore, the ${\color{blue} hd}$ constructor is necessary on the variable substitution equation in order produce an elimination: }

  \only<3>{ \[ [P/x]x = {\color{blue} hd}(P). \] }
  \onslide<4->{
    \[ \xymatrix@C=0pt{
        [& P & / & x &]x =& {\color{blue} hd}(P). \\
        & \hspace{-1em} \text{ term } \ar@{.}@/_1pc/[u] &
        & \text{ elim } \hspace{-1em} \ar@{.}@/^1pc/[u] &
        & \text{ not mere replacement } \hspace{-8em} \ar@{.}@/^1pc/[u]
      }
    \]
  }
  
\end{frame}
