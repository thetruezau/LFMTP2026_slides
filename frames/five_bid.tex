\begin{frame}
  \frametitle{System $\ulb$}
  
  \begin{definition}[Syntax of $\ulb$]
    \vspace{-1em} 
    \begin{align*}
      \text{(terms)} \hspace{-5em} &&\quad P,Q      &::= \lb x.t \mid \app{E} \\
      \text{(eliminations)} \hspace{-5em} &&\quad E &::= \only<1>{ x } \only<2->{ {\color{red} x} } \mid \hd{P} \mid EQ
    \end{align*}
    \vspace{-1em}
  \end{definition}
  
  \onslide<2->{ Here, the variable lives in the elimination class and is not a term. }

  \onslide<3->{ Therefore, the $hd$ constructor is necessary on the variable substitution equation in order produce an elimination:
    \[ [P/x]x = {\color{red} hd}(P). \] }
  \vspace{-1em}
  
  \onslide<4->{ As before, a substitution $\sub Px$ is replacing a variable $x$, which does not belong to the class of terms. }
  
\end{frame}
