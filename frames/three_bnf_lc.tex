\begin{frame}
  \frametitle{Hereditary substitution}

  % Another kind of unsupported substitution operation is the hereditary substitution.
  % \onslide<2->{ Take as an example the $\beta$-normal terms of the ordinary $\lb$-calculus in vectorial notation,
  \vspace{-1ex}
  \begin{definition}[Vectorial syntax of $\lb$-terms in $\beta$-nf]
    \[ M, N ::= \lb x.M \mid x \vec{M} \]
  \end{definition}
  
  \vspace{-1ex}
  \onslide<2->{
    Defining a substitution preserving $\beta$-normal forms leads to the equation\footnote<2->{$M'$ and $\vec{M}'$ abbreviate the recursive substitution calls},
    \[ \xymatrix{
        & \sub{(\lb y.N)}{x}{(x M \vec{M})} = 
          (\sub{M'}{x}{N}) @ \vec{M}', &
        \only<3->{ \text{Unconventional!} \ar@{.}@/^1pc/[ll] \ar@{.}@/_1pc/[ll] }
      }
    \]
    where one needs to continuously reduce the head term and properly aggregate the following applications (@ operation).
  }
  
  \onslide<4->{ This substitution operation is unconventional, as it is not a mere replacement of a free occurrence, and its non-structural definition does not assure termination. }
  
\end{frame}
